Person := \project_{ssn, name} Person_URA;
pk_{ssn} Person;

PersonPhone := \project_{ssn, phone} Person_URA;
pk_{ssn, phone} PersonPhone;

PersonEmail := \project_{ssn, email} Person_URA;
pk_{ssn, email} PersonEmail;

Employee := \project_{ssn, empid} \select_{defined(empid) and defined(hdate)} Person_URA;
pk_{empid} Employee;

EmployeeDate := \project_{empid, hdate} \select_{defined(empid) and defined(hdate)} Person_URA;
pk_{empid} EmployeeDate;

PED := \project_{ssn, empid} \select_{defined(empid) and defined(hdate) and defined(dept) and defined(manager)} Person_URA;
pk_{empid} PED;

PEDDept := \project_{empid, dept} \select_{defined(empid) and defined(hdate) and defined(dept) and defined(manager)} Person_URA;
pk_{empid} PEDDept;

DeptManager := \project_{dept, manager} \select_{defined(empid) and defined(hdate) and defined(dept) and defined(manager)} Person_URA;
pk_{dept} Dept_Manager;

inc=_{ssn, ssn} (Person, PersonPhone);
inc=_{ssn, ssn} (Person, PersonEmail);
inc=_{empid, empid} (Employee, EmployeeDate);
inc=_{empid, empid} (PED, PEDDept);
inc=_{dept, dept} (PEDDept, DeptManager);

inc⊆_{ssn, ssn} (Employee, Person);
inc⊆_{empid, empid} (PED, Employee);
inc⊆_{manager, empid} (DeptManager, Employee);
